```js
let requestURL = 'data.json';
let request = new XMLHttpRequest();
request.open("GET", requestURL, true);
request.onreadystatechange = () => {
    if (request.readyState == 4 && request.status == 200) {
        var data = JSON.parse(request.responseText);
        console.log(data);
    }
};
request.send();
```
